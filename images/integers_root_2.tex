\begin{tikzpicture}
  % Draw the number line
  \draw[->] (-3,0) -- (3,0);
  
  % Draw ticks and label the numbers
  \foreach \x in {-2,...,2} {
    \draw (\x,-0.1) -- (\x,0.1);
    \node[below] at (\x,-0.1) {$\x$};
  }
  
  % Add âˆš2 to the number line
  \draw[magenta, thick] ({sqrt(2)}, -0.2) -- ({sqrt(2)}, 0.2);
  \node[magenta, thick, above] at ({sqrt(2)}, 0.2) {$\sqrt{2}$};
\end{tikzpicture}
